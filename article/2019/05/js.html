  2. 加減、指定數值 
    3. var mydate = new Date("2018/11/24");
mydate.setFullYear(mydate.getFullYear()+1); // FullYear 為西元4碼
mydate.setMonth(mydate.getMonth()+1);       // Month 為 0 ~ 11
mydate.setDate(mydate.getDate()+1);         // Date 為 1 ~ 31
